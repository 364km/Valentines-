  <script>
    const noBtn = document.getElementById("noBtn");
    const yesBtn = document.getElementById("yesBtn");
    const letter = document.getElementById("letter");

    // NO button behavior: fly away when hover or click
    noBtn.addEventListener("mouseover", moveButton);
    noBtn.addEventListener("click", moveButton);

    function moveButton() {
      const x = Math.random() * window.innerWidth - window.innerWidth/2;
      const y = Math.random() * window.innerHeight - window.innerHeight/2;
      const scale = (Math.random() * 0.5 + 0.5).toFixed(2);
      noBtn.style.transform = `translate(${x}px, ${y}px) scale(${scale})`;
    }

    // YES button behavior: show big letter with animation
    yesBtn.addEventListener("click", () => {
      letter.classList.add("show");
      const lines = letter.querySelectorAll("p");
      lines.forEach((line, i) => {
        setTimeout(() => {
          line.style.opacity = 1;
          line.style.transform = "translateY(0)";
        }, i * 400); // delay for each line
      });
    });

    function closeLetter() {
      letter.classList.remove("show");
      const lines = letter.querySelectorAll("p");
      lines.forEach(line => {
        line.style.opacity = 0;
        line.style.transform = "translateY(20px)";
      });
    }

    // floating hearts
    setInterval(() => {
      const heart = document.createElement("div");
      heart.className = "heart";
      heart.innerText = "ðŸ’—";
      heart.style.left = Math.random() * 100 + "vw";
      heart.style.animationDuration = Math.random() * 3 + 4 + "s";
      document.body.appendChild(heart);
      setTimeout(() => heart.remove(), 7000);
    }, 500);
